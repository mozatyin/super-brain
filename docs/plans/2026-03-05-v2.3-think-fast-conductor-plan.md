# V2.3: Think Fast + Conductor Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Replace the fixed-phase Chatter (turns 1-7/8-14/15+) with a dynamic Conductor that selects conversation mode each turn based on ThinkFast signals and ThinkSlow accumulated questions.

**Architecture:** ThinkFast runs after every speaker response (rule-based, no LLM). Conductor reads ThinkFast + ThinkSlow outputs and returns a ConductorAction (listen/follow_thread/ask_incisive/push). Chatter receives the action and generates the next message accordingly. The existing ThinkSlow is enhanced to accumulate incisive questions alongside trait confidence.

**Tech Stack:** Python 3.12, pydantic, anthropic SDK, pytest

```python
"""Think Fast: Real-time signal detection from each exchange (V2.3).

Runs after every speaker response. Rule-based (no LLM call).
Detects: new facts, emotional shifts, contradictions, openings, info entropy.
"""

from __future__ import annotations

import re

from super_brain.models import ThinkFastResult


# Patterns for detecting factual information
_FACT_PATTERNS = [
    # Job/career
    (r"\b(?:i(?:'m| am)|i work(?:ed)?)\s+(?:a |as |at |in |for )(.+?)(?:\.|,|$)", "career"),
    (r"\b(?:my job|my work|my career|my company)\b.+?(?:\.|,|$)", "career"),
    # Location
    (r"\b(?:i live|i(?:'m| am) (?:from|in|based in)|moved to)\s+(.+?)(?:\.|,|$)", "location"),
    # Relationships
    (r"\b(?:my (?:wife|husband|partner|girlfriend|boyfriend|spouse))\b", "relationship"),
    (r"\b(?:my (?:kids?|children|son|daughter|baby))\b", "family"),
    # Education
    (r"\b(?:i (?:studied|majored|went to)|my degree|graduated from)\b.+?(?:\.|,|$)", "education"),
    # Hobbies
    (r"\b(?:i (?:love|enjoy|like) (?:to )?(?:\w+ing))\b", "hobby"),
    (r"\b(?:i(?:'ve| have) been (?:\w+ing))\b.+?(?:\.|,|$)", "hobby"),
]

# Patterns for detecting openings (unexplored topics)
_OPENING_PATTERNS = [
    r"\b(?:i(?:'ve| have) been (?:thinking|wanting|meaning) (?:about|to))\s+(.+?)(?:\.|,|$)",
    r"\b(?:i(?:'d| would) (?:love|like|want) to)\s+(.+?)(?:\.|,|$)",
    r"\b(?:haven't (?:started|tried|done|gotten around to))\b.+?(?:\.|,|$)",
    r"\b(?:someday|one day|eventually)\b.+?(?:\.|,|$)",
    r"\b(?:i wish|if only)\b.+?(?:\.|,|$)",
]


class ThinkFast:
    """Rule-based real-time signal detection from conversation exchanges."""

    def analyze(self, conversation: list[dict]) -> ThinkFastResult:
        """Analyze the last exchange in the conversation for signals."""
        if not conversation:
            return ThinkFastResult()

        # Focus on last speaker message
        last_speaker = None
        for msg in reversed(conversation):
            if msg["role"] == "speaker":
                last_speaker = msg["text"]
                break

        if not last_speaker:
            return ThinkFastResult()

        text = last_speaker
        text_lower = text.lower()

        # Detect new facts
        new_facts = self._detect_facts(text, text_lower)

        # Detect openings
        opening = self._detect_opening(text, text_lower)

        # Compute info entropy (simple: based on text length and content variety)
        info_entropy = self._compute_entropy(text, new_facts, opening)

        return ThinkFastResult(
            new_facts=new_facts,
            emotional_shift=None,  # V2.3: not yet implemented
            contradiction=None,    # V2.3: not yet implemented
            opening=opening,
            info_entropy=info_entropy,
        )

    def _detect_facts(self, text: str, text_lower: str) -> list[str]:
        """Detect factual information from speaker text."""
        facts = []
        for pattern, category in _FACT_PATTERNS:
            matches = re.finditer(pattern, text_lower, re.IGNORECASE)
            for match in matches:
                fact_text = match.group(0).strip().rstrip(".,")
                if len(fact_text) > 5:  # skip trivially short matches
                    facts.append(fact_text)
        return facts

    def _detect_opening(self, text: str, text_lower: str) -> str | None:
        """Detect unexplored topics hinted at by the speaker."""
        for pattern in _OPENING_PATTERNS:
            match = re.search(pattern, text_lower, re.IGNORECASE)
            if match:
                return match.group(0).strip().rstrip(".,")
        return None

    def _compute_entropy(
        self,
        text: str,
        facts: list[str],
        opening: str | None,
    ) -> float:
        """Compute information entropy of this exchange.

        Simple heuristic based on:
        - Text length (more words = more info)
        - Number of facts detected
        - Presence of an opening
        """
        words = text.split()
        word_count = len(words)

        # Base entropy from word count (normalized: 50 words = 0.5 entropy)
        base = min(1.0, word_count / 100.0)

        # Bonus for facts
        fact_bonus = min(0.3, len(facts) * 0.1)

        # Bonus for opening
        opening_bonus = 0.1 if opening else 0.0

        entropy = min(1.0, base + fact_bonus + opening_bonus)

        # Very short responses are always low entropy
        if word_count <= 5:
            entropy = min(entropy, 0.2)

        return round(entropy, 2)
```

```python
"""Conductor: Dynamic probabilistic action selection for conversation (V2.3).

Replaces the fixed phase system (turns 1-7/8-14/15+) with dynamic
decision-making based on ThinkFast and ThinkSlow signals.
"""

from __future__ import annotations

from super_brain.models import (
    ThinkFastResult,
    ThinkSlowResult,
    ConductorAction,
)


class Conductor:
    """Selects the next conversation action based on thinking agent signals."""

    def __init__(self, trust_building_turns: int = 3):
        self._trust_turns = trust_building_turns

    def decide(
        self,
        think_fast: ThinkFastResult,
        think_slow: ThinkSlowResult | None,
        turn_number: int,
    ) -> ConductorAction:
        """Select the next conversation action.

        Priority order:
        1. Early turns → always listen (build trust)
        2. ThinkFast detected an opening → follow it
        3. Info is stale + accumulated questions → ask top question
        4. Default → keep listening
        """
        # Priority 1: Early turns — build trust
        if turn_number <= self._trust_turns:
            return ConductorAction(
                mode="listen",
                context="Build rapport. Keep it warm and light.",
            )

        # Priority 2: Opening detected → follow it
        if think_fast.opening:
            return ConductorAction(
                mode="follow_thread",
                context=f"Follow up on: {think_fast.opening}",
            )

        # Priority 3: Info stale + questions accumulated → ask
        if (think_fast.info_entropy < 0.3
                and think_slow is not None
                and think_slow.incisive_questions):
            top_q = think_slow.incisive_questions[0]
            return ConductorAction(
                mode="ask_incisive",
                context=f"Target: {top_q.target} ({top_q.source})",
                question=top_q.question,
            )

        # Default: keep listening
        return ConductorAction(
            mode="listen",
            context="Continue deep listening. Follow their lead.",
        )
```

```python
def _generate_incisive_questions(
    low_confidence_traits: list[str],
    confidence_map: dict[str, float],
    max_questions: int = 10,
) -> list[IncisiveQuestion]:
    """Generate incisive questions from trait gaps.

    Prioritizes traits with lowest confidence. Uses trait_topic_map
    to find natural conversation topics for each trait.
    """
    # Sort low-confidence traits by confidence (lowest first)
    # Unestimated traits (not in confidence_map) get confidence 0.0
    sorted_traits = sorted(
        low_confidence_traits,

    for trait_name, topic in zip(sorted_traits, topics):
        questions.append(IncisiveQuestion(
            question=topic,
            target=trait_name,
            priority=1.0 - confidence_map.get(trait_name, 0.0),
            source="trait_gap",
        ))

    return questions[:max_questions]
```

```python
def _build_chatter_from_action(action: "ConductorAction") -> str:
    """Build Chatter system prompt from a ConductorAction (V2.3).

    Replaces the fixed-phase _build_chatter_system with dynamic,
    Conductor-driven prompt generation. All modes share the Deep Listening
    base principles but differ in their behavioral instruction.
    """
    from super_brain.models import ConductorAction

    base = (
        "You are a deep listener having a natural conversation. Your goal is to create "
        "a space where the other person feels genuinely heard and opens up naturally.\n\n"
        "DEEP LISTENING PRINCIPLES (follow these throughout):\n"
        "1. Full Attention & Presence — focus entirely on what they're saying, never rush\n"
        "2. Ease — no agenda, no pushing, let the conversation breathe\n"
        "3. Equality — treat them as an equal thinking partner, not a subject\n"
        "4. Appreciation — honor their openness genuinely\n"
        "5. Encouragement — gently invite deeper exploration only when they seem ready\n"
        "6. Feelings — all emotions are welcome, never judge or dismiss\n"
        "7. Place — create psychological safety through warmth and acceptance\n\n"
        "RESPONSE STYLE:\n"
        "- Keep your messages SHORT (1-2 sentences). Your job is to get THEM talking.\n"
        "- Ask ONE follow-up question per message, not multiple.\n"
        "- Reflect back what you heard before asking the next question.\n\n"
        "IMPORTANT: Do NOT probe their personality or ask them to describe themselves.\n\n"
    )

    if action.mode == "listen":
        return base + (
            "CURRENT MODE: Deep Listening.\n"
            f"{action.context}\n"
            "Follow their lead. Mirror their energy. Show genuine curiosity."
        )

    elif action.mode == "follow_thread":
        return base + (
            "CURRENT MODE: Follow Thread.\n"
            "The person just hinted at something interesting. Gently explore it.\n"
            f"{action.context}\n"
            "Don't push — just show curiosity. 'That sounds interesting, tell me more?'"
        )

    elif action.mode == "ask_incisive":
        prompt = base + (
            "CURRENT MODE: Incisive Question.\n"
            "There's a specific area we haven't explored yet. Weave this question "
            "naturally into the conversation — don't ask it out of nowhere.\n"
            f"{action.context}\n"
        )
        if action.question:
            prompt += (
                f"\nQUESTION TO WEAVE IN: \"{action.question}\"\n"
                "Adapt this question to fit the current conversation flow. "
                "Don't read it verbatim — make it sound natural."
            )
        return prompt

    elif action.mode == "push":
        prompt = base + (
            "CURRENT MODE: Gentle Push.\n"
            "You have enough rapport to ask a deeper, more challenging question.\n"
            f"{action.context}\n"
        )
        if action.question:
            prompt += (
                f"\nQUESTION: \"{action.question}\"\n"
                "Ask naturally. Frame it with curiosity, not interrogation."
            )
        return prompt

    # Fallback
    return base + f"CURRENT MODE: Listen.\n{action.context}"
```

```python
def simulate_conversation(
    chatter: Chatter,
    speaker: PersonalitySpeaker,
    profile: PersonalityDNA,
    n_turns: int,
    seed: int = 0,
    think_slow: "ThinkSlow | None" = None,
) -> "list[dict] | tuple[list[dict], list]":
    """Simulate a natural conversation for n_turns exchanges.

    V2.3: Uses ThinkFast + Conductor for dynamic action selection.
    ThinkSlow runs every 3 turns (adaptive). Conductor decides each turn's mode.
    """
    import random as _random
    rng = _random.Random(seed)

    from super_brain.think_fast import ThinkFast
    from super_brain.conductor import Conductor

    think_fast = ThinkFast()
    conductor = Conductor()

    conversation: list[dict] = []
    ts_results: list = []
    previous_ts = None
    last_think_fast = ThinkFastResult()

    # Start with a random casual opener
    opener = rng.choice(CASUAL_OPENERS)
    conversation.append({"role": "chatter", "text": opener})

    # Speaker responds
    reply = speaker.respond(profile, conversation, turn_number=0)
    conversation.append({"role": "speaker", "text": reply})

    # ThinkFast on first response
    if think_slow:
        last_think_fast = think_fast.analyze(conversation)

    # Continue for n_turns - 1 more exchanges
    think_slow_interval = 3  # V2.3: default every 3 turns
    for turn in range(1, n_turns):
        # Conductor decides action (V2.3)
        if think_slow:
            action = conductor.decide(
                think_fast=last_think_fast,
                think_slow=previous_ts,
                turn_number=turn + 1,
            )
            chatter_msg = chatter.next_message(
                conversation, turn_number=turn + 1, total_turns=n_turns,
                conductor_action=action,
            )
        else:
            chatter_msg = chatter.next_message(
                conversation, turn_number=turn + 1, total_turns=n_turns,
            )

        conversation.append({"role": "chatter", "text": chatter_msg})

        # Speaker responds in character
        speaker_reply = speaker.respond(profile, conversation, turn_number=turn)
        conversation.append({"role": "speaker", "text": speaker_reply})

        # ThinkFast after every speaker response (V2.3)
        if think_slow:
            last_think_fast = think_fast.analyze(conversation)

        # Think Slow extraction at adaptive intervals (V2.3: every 3 turns default)
        if think_slow and (turn + 1) % think_slow_interval == 0:
            focus = previous_ts.low_confidence_traits if previous_ts else None
            ts_result = think_slow.extract(
                conversation=conversation,
                focus_traits=focus,
                previous=previous_ts,
            )
            ts_results.append(ts_result)
            previous_ts = ts_result

    if think_slow is not None:
        return conversation, ts_results
    return conversation
```
